\section{Pendahuluan}
\begin{frame}{Latar Belakang}
  \begin{itemize}
    \item Kebutuhan sistem audio digital yang stabil dan presisi.
    \item Standardisasi antarmuka audio: I2S (Inter-IC Sound).
    \item Integrasi dengan SoC MicroBlaze V menggunakan AXI4-Lite.
    \item Validasi hardware dan perbaikan bug firmware.
  \end{itemize}
\end{frame}

\section{Dasar Teori}
\begin{frame}{Protokol I2S Philips Standard}
  \begin{itemize}
    \item Komunikasi tiga jalur: BCLK, WS (LRCK), DATA.
    \item Format frame stereo: 64 BCLK per frame.
    \item Data berubah pada tepi turun, dibaca pada tepi naik.
  \end{itemize}
\end{frame}

\section{Perancangan Sistem}
\begin{frame}{Arsitektur IP I2S TX}
  \begin{itemize}
    \item Integrasi AXI4-Lite Slave.
    \item Pengendali clock (audio\_clk) dengan divisor.
    \item Serializer data MSB-first.
  \end{itemize}
\end{frame}

\section{Implementasi Firmware}
\begin{frame}{Perbaikan Bug Firmware}
  \begin{itemize}
    \item Koreksi masker bit sampel pada C.
    \item Perbaikan presisi timing loop menggunakan \texttt{rdcycle64()}.
  \end{itemize}
\end{frame}

\section{Penutup}
\begin{frame}{Kesimpulan}
  \begin{itemize}
    \item IP I2S TX berhasil diimplementasikan.
    \item Bug firmware berhasil diidentifikasi dan diperbaiki tanpa mengubah RTL.
  \end{itemize}
\end{frame}
