% 3.3  PERANCANGAN REGISTER MAP AXI4-LITE
%   Jelaskan lima register 32-bit yang dipetakan pada ruang alamat:
%     0x00 – CTRL    : [1] IRQ_EN, [0] TX_EN
%     0x04 – MODE    : [0] FS_SEL (0=Mode0~48k, 1=Mode1~96k)
%     0x08 – TX_DATA : [23:0] audio_sample (tulis ke FIFO, alternating L/R)
%     0x0C – FIFO_STAT: [9] FIFO_FULL, [8] FIFO_EMPTY, [7:0] FIFO_LEVEL
%     0x10 – FIFO_THR : [7:0] watermark threshold (default 0x40 = 64)
%     0x14 – IRQ_STAT : [0] WM_IRQ (tulis 1 untuk clear)
%
% [TABLE PLACEHOLDER]
%   Tabel 3.2 – Register map AXI4-Lite peripheral I2S TX
%   Kolom: Offset | Nama | Bit | Akses | Deskripsi | Reset
% ================================================================
\section{Perancangan Register Map AXI4-Lite}

% [TODO: tulis narasi singkat tentang prinsip register map]

% [TABLE PLACEHOLDER]
\begin{table}[ht]
  \centering
  \caption{Register map AXI4-Lite peripheral I2S TX.}
  \label{tab:regmap}
  % [TODO: isi deskripsi dan nilai reset setiap register]
  \begin{tabular}{cllllp{4cm}}
    \hline
    Offset & Nama & Bit & Akses & Reset & Deskripsi \\
    \hline
    0x00 & CTRL      & [1:0]  & R/W & 0x0  &
      [1] IRQ\_EN: aktifkan watermark IRQ; [0] TX\_EN: mulai transmisi \\
    0x04 & MODE      & [0]    & R/W & 0x0  &
      [0] FS\_SEL: 0~=~Mode~0 ($\approx$48~kHz), 1~=~Mode~1 ($\approx$96~kHz) \\
    0x08 & TX\_DATA  & [23:0] & W   & --   &
      Tulis 24-bit audio sample; setiap dua tulis = 1 frame L+R \\
    0x0C & FIFO\_STAT & [9:0] & R   & 0x0  &
      [9] FULL, [8] EMPTY, [7:0] level (jumlah frame terisi) \\
    0x10 & FIFO\_THR & [7:0]  & R/W & 0x40 &
      Watermark threshold dalam satuan frame (default 64) \\
    0x14 & IRQ\_STAT & [0]    & R/W1C & 0x0 &
      [0] WM\_IRQ: set saat FIFO\_LEVEL $\leq$ FIFO\_THR; tulis 1 untuk clear \\
    \hline
  \end{tabular}
\end{table}
