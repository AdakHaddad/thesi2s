\chapter*{Ringkasan IP Block Design}
\addcontentsline{toc}{chapter}{Ringkasan IP Block Design}

Lampiran ini merangkum IP utama pada Vivado Block Design \path{bdesign_clock_core}
yang berada pada proyek \path{nobufgmux}. Berkas deskripsi block design utama
tersimpan pada \path{nobufgmux/nobufgmux.srcs/sources_1/bd/bdesign_clock_core/bdesign_clock_core.bd},
sedangkan hasil generasi IP berada pada
\path{nobufgmux/nobufgmux.gen/sources_1/bd/bdesign_clock_core/ip/}.

\section*{Ringkasan arsitektur}
\begin{itemize}
    \item Clock sistem berasal dari \path{sys_clock} 100 MHz dan masuk ke \path{clk_wiz_0}.
    \item \path{clk_wiz_0} menghasilkan tiga keluaran: 100 MHz untuk domain sistem, 24{,}576 MHz untuk keluarga 48/96 kHz, dan 22{,}579 MHz untuk keluarga 44{,}1/88{,}2 kHz.
    \item \path{microblaze_riscv_0} bertindak sebagai master AXI dan terhubung ke \path{axi_smc}.
    \item \path{axi_smc} merutekan transaksi ke dua slave: \path{i2s_0} dan \path{axi_uartlite_0}.
    \item Keluaran audio eksternal yang diekspos block design adalah \path{i2s_mclk_0}, \path{i2s_bclk_0}, \path{i2s_ws_0}, dan \path{i2s_data_0}.
\end{itemize}

\section*{\path{clk_wiz_0}}
\begin{description}
    \item[Purpose] Clocking Wizard untuk membangkitkan domain clock sistem dan dua keluarga clock audio.
    \item[Key parameters] \path{CLK_IN1=100 MHz}, \path{CLKOUT2=24.576 MHz}, \path{CLKOUT3=22.579 MHz}.
    \item[Connections] \path{clk_out1} ke domain 100 MHz; \path{clk_out2} ke \path{i2s_0/audio_48_clk}; \path{clk_out3} ke \path{i2s_0/audio_44_clk}.
\end{description}

\section*{\path{microblaze_riscv_0}}
\begin{description}
    \item[Purpose] Soft processor RISC-V untuk menjalankan firmware \path{micblaze/ws/rtlfix/src/helloworld.c}.
    \item[Key parameters] Debug aktif, jalur \path{M_AXI_DP} untuk akses peripheral, serta ILMB/DLMB untuk memori lokal.
    \item[Connections] Clock 100 MHz dari \path{clk_wiz_0}; reset dari \path{rst_clk_wiz_0_100M}; master AXI ke \path{axi_smc/S00_AXI}.
\end{description}

\section*{\path{microblaze_riscv_0_local_memory}}
\begin{description}
    \item[Purpose] Subsistem memori lokal instruksi dan data.
    \item[Sub-IP] \path{dlmb_v10}, \path{ilmb_v10}, \path{dlmb_bram_if_cntlr}, \path{ilmb_bram_if_cntlr}, dan \path{lmb_bram}.
    \item[Key parameters] BRAM bertipe \path{True_Dual_Port_RAM} dan dihubungkan ke kedua jalur LMB.
\end{description}

\section*{\path{mdm_1}}
\begin{description}
    \item[Purpose] \textit{Microprocessor Debug Module} untuk debug prosesor.
    \item[Connections] Antarmuka debug langsung ke \path{microblaze_riscv_0/DEBUG}.
\end{description}

\section*{\path{rst_clk_wiz_0_100M}}
\begin{description}
    \item[Purpose] Reset controller sinkron untuk domain 100 MHz.
    \item[Connections] Menghasilkan \path{mb_reset}, \path{bus_struct_reset}, dan \path{peripheral_aresetn} untuk MicroBlaze, SmartConnect, I2S, dan UARTLite.
\end{description}

\section*{\path{axi_smc}}
\begin{description}
    \item[Purpose] AXI SmartConnect tunggal yang menghubungkan satu master ke dua slave.
    \item[Connections] \path{S00_AXI} dari \path{microblaze_riscv_0}; \path{M00_AXI} ke \path{i2s_0}; \path{M01_AXI} ke \path{axi_uartlite_0}.
\end{description}

\section*{\path{i2s_0}}
\begin{description}
    \item[Purpose] Custom IP I2S transmitter dari \path{ip_repo/i2s_1_0}.
    \item[Main interfaces] \path{S00_AXI} untuk kontrol register, \path{audio_48_clk}, \path{audio_44_clk}, dan empat keluaran I2S.
    \item[Key parameters] Tiga register AXI4-Lite 32-bit, \path{C_S00_AXI_BASEADDR=0x00020000}, domain data 100 MHz dan domain audio terpisah.
\end{description}

\section*{\path{axi_uartlite_0}}
\begin{description}
    \item[Purpose] UARTLite untuk antarmuka terminal firmware.
    \item[Main interfaces] \path{S_AXI} untuk konfigurasi AXI dan \path{UART} untuk \path{usb_uart_txd}/\path{usb_uart_rxd}.
    \item[Key parameters] Alamat basis \path{0x04060000} pada ruang alamat MicroBlaze.
\end{description}
