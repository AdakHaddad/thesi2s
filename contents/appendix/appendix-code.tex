\subsection{Modul Utama I2S Peripheral (\textit{Top Module})}

\begin{lstlisting}[language=Verilog, caption={Kerangka modul utama peripheral I2S TX-only
  dengan antarmuka AXI4-Lite. Parameter dan port lengkap; logika internal diisi pada
  implementasi.}]
module i2s_tx_axi #(
    parameter integer C_S_AXI_DATA_WIDTH  = 32,
    parameter integer C_S_AXI_ADDR_WIDTH  = 8,   // 256 byte = 64 register 32-bit
    parameter integer FIFO_DEPTH          = 256   // frame stereo
)(
    // ---- AXI4-Lite Slave Interface ----
    input  wire                            S_AXI_ACLK,
    input  wire                            S_AXI_ARESETN,
    // Write address channel
    input  wire [C_S_AXI_ADDR_WIDTH-1:0]  S_AXI_AWADDR,
    input  wire                            S_AXI_AWVALID,
    output wire                            S_AXI_AWREADY,
    // Write data channel
    input  wire [C_S_AXI_DATA_WIDTH-1:0]  S_AXI_WDATA,
    input  wire [(C_S_AXI_DATA_WIDTH/8)-1:0] S_AXI_WSTRB,
    input  wire                            S_AXI_WVALID,
    output wire                            S_AXI_WREADY,
    // Write response channel
    output wire [1:0]                      S_AXI_BRESP,
    output wire                            S_AXI_BVALID,
    input  wire                            S_AXI_BREADY,
    // Read address channel
    input  wire [C_S_AXI_ADDR_WIDTH-1:0]  S_AXI_ARADDR,
    input  wire                            S_AXI_ARVALID,
    output wire                            S_AXI_ARREADY,
    // Read data channel
    output wire [C_S_AXI_DATA_WIDTH-1:0]  S_AXI_RDATA,
    output wire [1:0]                      S_AXI_RRESP,
    output wire                            S_AXI_RVALID,
    input  wire                            S_AXI_RREADY,

    // ---- Sumber clock audio (dari MMCM clk_out2 ~24.57 MHz) ----
    input  wire                            audio_clk,

    // ---- Antarmuka I2S ke PCM5102A ----
    output wire                            i2s_bclk,
    output wire                            i2s_ws,
    output wire                            i2s_data,

    // ---- Interupsi ke AXI Interrupt Controller ----
    output wire                            irq
);

// ---- Offset register (byte address) ----
localparam ADDR_ID         = 8'h00;
localparam ADDR_CONTROL    = 8'h04;
localparam ADDR_STATUS     = 8'h08;
localparam ADDR_WATERMARK  = 8'h0C;
localparam ADDR_IRQ_EN     = 8'h10;
localparam ADDR_IRQ_STATUS = 8'h14;
localparam ADDR_VOLUME     = 8'h18;
localparam ADDR_TX_LEFT    = 8'h1C;
localparam ADDR_TX_RIGHT   = 8'h20;

// ---- Register internal ----
reg [31:0] reg_control;     // RW
reg [31:0] reg_watermark;   // RW
reg [31:0] reg_irq_en;      // RW
reg [31:0] reg_irq_status;  // W1C
reg [31:0] reg_volume;      // RW
reg [23:0] reg_tx_left;     // WO (latch sementara)

// ---- Sinyal FIFO ----
// ... (deklarasi sinyal FIFO: wr_en, rd_en, din, dout, level, empty, full)

// ---- AXI4-Lite slave logic ----
// ... (implementasi AXI handshake: AWREADY, WREADY, BVALID, ARREADY, RVALID)
// ... (decode alamat tulis: update register sesuai ADDR_*)
// ... (decode alamat baca: mux RDATA sesuai ADDR_*)
// ... (khusus TX_RIGHT: push {reg_tx_left, wdata[23:0]} ke FIFO)

// ---- TX FIFO (array register sinkron) ----
// ... (implementasi FIFO: head/tail pointer, level counter, underrun flag)

// ---- I2S Serializer ----
// ... (generator BCLK dari audio_clk, divisor tergantung FS_MODE)
// ... (generator WS: toggle setiap 32 BCLK; delay 1 BCLK sebelum data)
// ... (shift register 32-bit: load dari FIFO pada awal frame, shift MSB-first)
// ... (penanganan underrun: output 0 jika FIFO kosong)

// ... (assert irq jika fifo_level <= reg_watermark && irq_en[0])
// ... (assert irq jika underrun && irq_en[1])

\section{Aplikasi Demo Vitis (\texttt{main.cpp})}

\begin{lstlisting}[language=C, caption={Kerangka aplikasi demo bare-metal \texttt{main.cpp}
  untuk pemutaran sinyal sinus 1\,kHz melalui peripheral I2S menggunakan ISR
  watermark pada Xilinx Vitis/MicroBlaze.}]
/* main.cpp -- I2S TX Demo (bare-metal MicroBlaze, Xilinx Vitis)
 * Platform : Basys3 (Artix-7 XC7A35T)
 * Peripheral: I2S TX kustom dengan AXI4-Lite
 * Output   : sinyal sinus 1 kHz melalui DAC PCM5102A
 */

#include "xparameters.h"
#include "xil_io.h"
#include "xil_printf.h"
#include "xintc.h"
#include "microblaze_sleep.h"
#include <math.h>
#include <stdint.h>

/* ---- Alamat dasar peripheral (dari xparameters.h setelah export hardware) ---- */
#define I2S_BASE    XPAR_I2S_TX_AXI_0_S_AXI_BASEADDR
#define INTC_BASE   XPAR_INTC_0_BASEADDR
#define I2S_IRQ_ID  XPAR_INTC_0_I2S_TX_AXI_0_VEC_ID

/* ---- Offset register I2S ---- */
#define I2S_ID          (I2S_BASE + 0x00)
#define I2S_CONTROL     (I2S_BASE + 0x04)
#define I2S_STATUS      (I2S_BASE + 0x08)
#define I2S_WATERMARK   (I2S_BASE + 0x0C)
#define I2S_IRQ_EN      (I2S_BASE + 0x10)
#define I2S_IRQ_STATUS  (I2S_BASE + 0x14)
#define I2S_VOLUME      (I2S_BASE + 0x18)
#define I2S_TX_LEFT     (I2S_BASE + 0x1C)
#define I2S_TX_RIGHT    (I2S_BASE + 0x20)

/* ---- Bit definisi register CONTROL ---- */
#define CTRL_ENABLE     (1 << 0)
#define CTRL_MUTE       (1 << 1)
#define CTRL_FIFO_RST   (1 << 2)
#define CTRL_CLR_URUN   (1 << 3)
#define CTRL_FS_96K     (1 << 4)  /* 1 = ~96kHz; 0 = ~48kHz */

/* ---- Parameter audio ---- */
#define SAMPLE_RATE_96K  95972    /* Hz aktual mode 1 */
#define SINE_FREQ        1000     /* Hz */
#define SINE_SAMPLES     96       /* satu periode pada 96kHz */
#define AMPLITUDE_24BIT  0x600000 /* ~75% dari 2^23 */
#define FIFO_PREFILL     200      /* frame praisi sebelum ENABLE */

/* ---- Tabel sinyal sinus (24-bit, satu periode) ---- */
static int32_t sine_table[SINE_SAMPLES];

/* ---- Posisi pembacaan tabel dalam ISR ---- */
static volatile uint32_t table_pos = 0;

/* ---- Instance INTC ---- */
static XIntc intc;

/* Fungsi: bangkitkan tabel sinyal sinus 24-bit */
static void generate_sine_table(void) {
    int i;
    for (i = 0; i < SINE_SAMPLES; i++) {
        sine_table[i] = (int32_t)(AMPLITUDE_24BIT *
                        sin(2.0 * M_PI * i / SINE_SAMPLES));
    }
}

/* Fungsi: tulis satu frame stereo (kiri + kanan) ke FIFO peripheral */
static inline void write_frame(int32_t left, int32_t right) {
    Xil_Out32(I2S_TX_LEFT,  (uint32_t)(left  & 0x00FFFFFF));
    Xil_Out32(I2S_TX_RIGHT, (uint32_t)(right & 0x00FFFFFF));
}

/* ISR: dipanggil saat FIFO level <= WATERMARK */
static void i2s_isr(void *callback) {
    uint32_t status = Xil_In32(I2S_STATUS);
    uint32_t level  = (status >> 8) & 0xFF;  /* FIFO_LEVEL */

    /* Isi FIFO hingga ~200 frame */
    while (level < 200) {
        int32_t sample = sine_table[table_pos];
        write_frame(sample, sample);  /* stereo: L = R (mono sine) */
        table_pos = (table_pos + 1) % SINE_SAMPLES;
        level++;
    }

    /* Clear IRQ_STATUS bit[0] (watermark) */
    Xil_Out32(I2S_IRQ_STATUS, 0x01);
}

int main(void) {
    int status;
    uint32_t i;

    xil_printf("\r\n=== I2S TX Demo: 1kHz Sine Wave @ ~96kHz ===\r\n");

    /* 1. Bangkitkan tabel sinus */
    generate_sine_table();
    xil_printf("Sine table generated (%d samples).\r\n", SINE_SAMPLES);

    /* 2. Reset FIFO dan nonaktifkan output */
    Xil_Out32(I2S_CONTROL, CTRL_FIFO_RST);
    MB_Sleep(1);  /* tunggu reset selesai */

    /* 3. Konfigurasi WATERMARK dan VOLUME */
    Xil_Out32(I2S_WATERMARK, 64);   /* IRQ saat level <= 64 frame */
    Xil_Out32(I2S_VOLUME,    0);    /* volume penuh */

    /* 4. Praisi FIFO dengan FIFO_PREFILL frame */
    for (i = 0; i < FIFO_PREFILL; i++) {
        int32_t sample = sine_table[i % SINE_SAMPLES];
        write_frame(sample, sample);
    }
    table_pos = FIFO_PREFILL % SINE_SAMPLES;
    xil_printf("FIFO prefilled: %d frames.\r\n", FIFO_PREFILL);

    /* 5. Inisialisasi dan daftarkan ISR ke INTC */
    status = XIntc_Initialize(&intc, INTC_BASE);
    if (status != XST_SUCCESS) {
        xil_printf("INTC init failed!\r\n");
        return -1;
    }
    XIntc_Connect(&intc, I2S_IRQ_ID, i2s_isr, NULL);
    XIntc_Enable(&intc, I2S_IRQ_ID);
    XIntc_Start(&intc, XIN_REAL_MODE);
    microblaze_enable_interrupts();

    /* 6. Aktifkan IRQ watermark dan enable peripheral (mode ~96kHz) */
    Xil_Out32(I2S_IRQ_EN,  0x01);                /* watermark IRQ enable */
    Xil_Out32(I2S_CONTROL, CTRL_ENABLE | CTRL_FS_96K);

    xil_printf("I2S enabled. Mode: ~96kHz. Playing 1kHz sine...\r\n");

    /* 7. Loop utama: cetak status setiap ~1 detik */
    while (1) {
        uint32_t st  = Xil_In32(I2S_STATUS);
        uint32_t lvl = (st >> 8) & 0xFF;
        xil_printf("FIFO level: %3u | underrun: %u | irq: %u\r\n",
                   lvl,
                   (st >> 2) & 1,
                   Xil_In32(I2S_IRQ_STATUS) & 1);
        MB_Sleep(1000);
    }

    return 0;
}
