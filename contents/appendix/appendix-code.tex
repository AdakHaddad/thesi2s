Berikut adalah kode sumber utama (\texttt{helloworld.c}) yang digunakan untuk menguji peripheral I2S pada platform Vitis bare-metal.

\begin{lstlisting}[language=C, caption=Kode sumber helloworld.c untuk pengujian I2S]
/*****************************************************************************
* I2S register feature test - all control bits
*****************************************************************************/

#include <stdint.h>
#include "platform.h"
#include "xil_printf.h"
#include "xil_io.h"
#include "xparameters.h"
#include "xuartlite_l.h"

/* Read the RISC-V cycle counter (mcycle CSR) */
static inline uint64_t rdcycle64(void)
{
    uint32_t lo, hi, hi2;
    do {
        __asm__ volatile ("rdcycleh %0" : "=r"(hi));
        __asm__ volatile ("rdcycle  %0" : "=r"(lo));
        __asm__ volatile ("rdcycleh %0" : "=r"(hi2));
    } while (hi != hi2);
    return ((uint64_t)hi << 32) | lo;
}

/* ---- Register map ---- */
#define I2S_BASE       XPAR_I2S_0_BASEADDR
#define DATA_LEFT      (I2S_BASE + 0x00U)
#define DATA_RIGHT     (I2S_BASE + 0x04U)
#define CONTROL        (I2S_BASE + 0x08U)

/* ---- Control bits ---- */
#define CTRL_ENABLE          (1U << 0)
#define CTRL_MUTE            (1U << 1)
#define CTRL_FS              (1U << 2)
#define CTRL_FS_MODE         (1U << 3)
#define CTRL_WIDTH(w)        (((uint32_t)(w) & 0x1FU) << 8U)

/* ---- Audio ---- */
#define SAMPLE_RATE    48000U
#define AMPLITUDE      32000

/* ---- Sine table ---- */
static const int16_t SINE[32] = {
         0,   6393,  12539,  18204,  23170,  27245,  30273,  32137,
     32767,  32137,  30273,  27245,  23170,  18204,  12539,   6393,
         0,  -6393, -12539, -18204, -23170, -27245, -30273, -32137,
    -32767, -32137, -30273, -27245, -23170, -18204, -12539,  -6393
};

/* ---- State ---- */
static uint32_t g_freq    = 1000U;
static uint8_t  g_left    = 1U;
static uint8_t  g_right   = 1U;
static uint8_t  g_enable  = 1U;
static uint8_t  g_mute    = 0U;
static uint8_t  g_fs      = 0U;
static uint8_t  g_fs_mode = 0U;
static uint8_t  g_width   = 24U;

static uint32_t current_sample_rate(void)
{
    if (!g_fs && !g_fs_mode) return 48000U;
    if ( g_fs && !g_fs_mode) return 44100U;
    if (!g_fs &&  g_fs_mode) return 96000U;
    return 88200U;
}

static void apply_control(void)
{
    uint32_t ctrl = 0U;
    if (g_enable)  ctrl |= CTRL_ENABLE;
    if (g_mute)    ctrl |= CTRL_MUTE;
    if (g_fs)      ctrl |= CTRL_FS;
    if (g_fs_mode) ctrl |= CTRL_FS_MODE;
    ctrl |= CTRL_WIDTH(g_width);
    Xil_Out32(CONTROL, ctrl);
}

static void stream_sample(void)
{
    static uint32_t phase = 0U;
    uint32_t sr   = current_sample_rate();
    uint32_t step = (uint32_t)(((uint64_t)g_freq << 32) / sr);

    int16_t s = (int16_t)(((int32_t)SINE[phase >> 27] * AMPLITUDE) / 32767);

    int32_t s32;
    if (g_width >= 32U)       s32 = (int32_t)s << 16;
    else if (g_width > 16U)   s32 = (int32_t)s << (g_width - 16U);
    else if (g_width < 16U)   s32 = (int32_t)s >> (16U - g_width);
    else                      s32 = (int32_t)s;
    
    uint32_t samp = (uint32_t)s32;

    Xil_Out32(DATA_LEFT,  g_left  ? samp : 0U);
    Xil_Out32(DATA_RIGHT, g_right ? samp : 0U);
    phase += step;
}

int main(void)
{
    init_platform();
    apply_control();

    uint64_t t_deadline = rdcycle64();

    while (1) {
        uint32_t sample_rate   = current_sample_rate();
        uint64_t period_cycles = (uint64_t)XPAR_CPU_CORE_CLOCK_FREQ_HZ / sample_rate;

        stream_sample();

        if (!XUartLite_IsReceiveEmpty(STDIN_BASEADDRESS)) {
            uint8_t ch = XUartLite_RecvByte(STDIN_BASEADDRESS);
            // Handle keypress logic...
        }

        t_deadline += period_cycles;
        while (rdcycle64() < t_deadline) {}
    }

    cleanup_platform();
    return 0;
}
\end{lstlisting}

// Glitch-free clock multiplexer
module clk_mux (
    input  wire clk0, clk1, sel,
    output wire clk_o
);
    reg q0_d1, q0_d2, q1_d1, q1_d2;
    always @(negedge clk0) begin
        q0_d1 <= !sel & !q1_d2;
        q0_d2 <= q0_d1;
    end
    always @(negedge clk1) begin
        q1_d1 <= sel & !q0_d2;
        q1_d2 <= q1_d1;
    end
    assign clk_o = (q0_d2 & clk0) | (q1_d2 & clk1);
endmodule

// Clock Selection Logic
    wire reg_fs_family  = slv_reg2[2];

    clk_mux mclk_mux_inst (
        .clk0  (audio_48_clk),

// CDC Sync
    cdc_sync #(.WIDTH(32)) cdc_ctrl_inst (

// Serializer function
    function i2s_next_serial_bit;
        input [31:0] left_samp;
        input [31:0] right_samp;
        input [5:0]  bc;
        input [5:0]  width;
        reg          in_left;
        reg  [4:0]   pos;
        reg  [5:0]   bit_idx;
        begin
            in_left = (bc < 6'd32);
            pos     = bc[4:0];
            if (width == 6'd0)
                i2s_next_serial_bit = 1'b0;
            else if (pos == 5'd0)
                i2s_next_serial_bit = 1'b0;
            else if (pos <= width) begin
                bit_idx = width - {1'b0, pos};
                i2s_next_serial_bit = in_left ? left_samp[bit_idx[4:0]]
                                              : right_samp[bit_idx[4:0]];
            end else
                i2s_next_serial_bit = 1'b0;
        end
    endfunction

// I2S output generator
    always @(posedge internal_mclk or posedge audio_rst) begin
        if (audio_rst) begin
            bit_count_q  <= 6'd0;
            data_q       <= 1'b0;
            ws_q         <= 1'b0;
            bclk_q       <= 1'b0;
        end else if (bclk_en_w) begin
            if (!enable_sync) begin
                bit_count_q  <= 6'd0;
                data_q       <= 1'b0;
                ws_q         <= 1'b0;
                bclk_q       <= 1'b0;
            end else begin
                if (bclk_q) begin
                    bclk_q      <= 1'b0;
                    data_q      <= i2s_next_serial_bit(
                                       sample_for_i2s_left,
                                       sample_for_i2s_right,
                                       bit_count_q,
                                       sample_width_sync);
                    ws_q        <= bit_count_q[5];
                    bit_count_q <= bit_count_q + 6'd1;
                end else begin
                    bclk_q <= 1'b1;
                end
            end
        end
    end
