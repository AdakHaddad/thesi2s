\section{Cuplikan kode implementasi I2S}

\subsection{Cuplikan RTL: sinkronisasi register AXI ke domain \texttt{audio\_clk}}

\begin{lstlisting}[language=Verilog]
always @(posedge audio_clk or posedge audio_rst)
begin
    if (audio_rst)
    begin
        ctrl_sync_a   <= 32'd0;
        ctrl_sync_b   <= 32'd0;
        sample_left_q <= 32'd0;
        sample_right_q<= 32'd0;
    end
    else
    begin
        ctrl_sync_a   <= slv_reg2;   // CONTROL @ 0x08
        ctrl_sync_b   <= ctrl_sync_a;
        sample_left_q <= slv_reg0;   // DATA_LEFT  @ 0x00
        sample_right_q<= slv_reg1;   // DATA_RIGHT @ 0x04
    end
end
\end{lstlisting}

\subsection{Cuplikan RTL: generator sinyal I2S Philips}

\begin{lstlisting}[language=Verilog]
wire bclk_en_w = fs_mode_sync ? (clock_div_q[0] == 1'b0)
                               : (clock_div_q[1:0] == 2'd0);

always @(posedge audio_clk or posedge audio_rst)
begin
    if (audio_rst)
    begin
        bit_count_q <= 6'd0;
        data_q      <= 1'b0;
        ws_q        <= 1'b0;
        bclk_q      <= 1'b0;
    end
    else if (bclk_en_w)
    begin
        if (!enable_sync)
        begin
            bit_count_q <= 6'd0;
            data_q      <= 1'b0;
            ws_q        <= 1'b0;
            bclk_q      <= 1'b0;
        end
        else if (bclk_q)
        begin
            bclk_q      <= 1'b0;
            data_q      <= i2s_next_serial_bit(sample_for_i2s_left,
                                               sample_for_i2s_right,
                                               bit_count_q,
                                               sample_width_sync);
            ws_q        <= bit_count_q[5]; // 0=left slot, 1=right slot
            bit_count_q <= bit_count_q + 6'd1;
        end
        else
            bclk_q <= 1'b1;
    end
end
\end{lstlisting}

\subsection{Cuplikan aplikasi uji Vitis (bare-metal)}

\begin{lstlisting}[language=C]
#define I2S_BASE      XPAR_I2S_0_BASEADDR
#define I2S_DATA_L    (I2S_BASE + 0x00U)
#define I2S_DATA_R    (I2S_BASE + 0x04U)
#define I2S_CONTROL   (I2S_BASE + 0x08U)

#define CTRL_ENABLE        (1U << 0)
#define CTRL_MUTE          (1U << 1)
#define CTRL_FS_MODE       (1U << 2)
#define CTRL_FS_FAMILY     (1U << 3)
#define CTRL_SAMPLE_WIDTH(w) (((w) & 0x1FU) << 8)

static inline void i2s_write_sample(u32 left, u32 right) {
    Xil_Out32(I2S_DATA_L, left);
    Xil_Out32(I2S_DATA_R, right);
}

int main(void) {
    init_platform();

    // ENABLE=1, MUTE=0, FS_MODE=0, FS_FAMILY=0, SAMPLE_WIDTH=24
    Xil_Out32(I2S_CONTROL, CTRL_ENABLE | CTRL_SAMPLE_WIDTH(24));

    while (1) {
        i2s_write_sample(0x00007FFF, 0x00007FFF);
        usleep(1000000U / 48000U);
    }
}
\end{lstlisting}
